128 bool FGPropulsion::Run(bool Holding)
129 {
130  unsigned int i;
131 
132  if (FGModel::Run(Holding)) return true;
133  if (Holding) return false;
134 
135  RunPreFunctions();
136 
137  vForces.InitMatrix();
138  vMoments.InitMatrix();
139 
140  for (i=0; i<numEngines; i++) {
141  Engines[i]->Calculate();
142  ConsumeFuel(Engines[i]);
143  vForces += Engines[i]->GetBodyForces(); // sum body frame forces
144  vMoments += Engines[i]->GetMoments(); // sum body frame moments
145  }
146 
147  TotalFuelQuantity = 0.0;
148  TotalOxidizerQuantity = 0.0;
149  for (i=0; i<numTanks; i++) {
150  Tanks[i]->Calculate( in.TotalDeltaT, in.TAT_c);
151  switch (Tanks[i]->GetType()) {
152  case FGTank::ttFUEL:
153  TotalFuelQuantity += Tanks[i]->GetContents();
154  break;
155  case FGTank::ttOXIDIZER:
156  TotalOxidizerQuantity += Tanks[i]->GetContents();
157  break;
158  default:
159  break;
160  }
161  }
162 
163  if (refuel.node() && refuel) DoRefuel( in.TotalDeltaT );
164  if (dump.node() && dump) DumpFuel( in.TotalDeltaT );
165 
166  RunPostFunctions();
167 
168  return false;
169 }

171 //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
172 //
173 // The engine can tell us how much fuel it needs, but it is up to the propulsion
174 // subsystem manager class FGPropulsion to manage fuel flow amongst tanks. Engines
175 // May burn fuel from more than one tank at a time, and may burn from one tank
176 // before another - that is, may burn from one tank until the tank is depleted,
177 // then burn from the next highest priority tank. This can be accompished
178 // by defining a fuel management system, but this way of specifying priorities
179 // is more automatic from a user perspective.
180 
181 void FGPropulsion::ConsumeFuel(FGEngine* engine)
182 {
183  if (FuelFreeze) return;
184  if (FDMExec->GetTrimStatus()) return;
185 
186  unsigned int TanksWithFuel=0, CurrentFuelTankPriority=1;
187  unsigned int TanksWithOxidizer=0, CurrentOxidizerTankPriority=1;
188  vector <int> FeedListFuel, FeedListOxi;
189  bool Starved = true; // Initially set Starved to true. Set to false in code below.
190  bool hasOxTanks = false;
191 
192  // For this engine,
193  // 1) Count how many fuel tanks with the current priority level have fuel
194  // 2) If there none, then try next lower priority (higher number) - that is,
195  // increment CurrentPriority.
196  // 3) Build the feed list.
197  // 4) Do the same for oxidizer tanks, if needed.
198 
199  // Process fuel tanks, if any
200  while ((TanksWithFuel == 0) && (CurrentFuelTankPriority <= numTanks)) {
201  for (unsigned int i=0; i<engine->GetNumSourceTanks(); i++) {
202  unsigned int TankId = engine->GetSourceTank(i);
203  FGTank* Tank = Tanks[TankId];
204  unsigned int TankPriority = Tank->GetPriority();
205  if (TankPriority != 0) {
206  switch(Tank->GetType()) {
207  case FGTank::ttFUEL:
208  if ((Tank->GetContents() > Tank->GetUnusable()) && Tank->GetSelected() && (TankPriority == CurrentFuelTankPriority)) {
209  TanksWithFuel++;
210  Starved = false;
211  FeedListFuel.push_back(TankId);
212  }
213  break;
214  case FGTank::ttOXIDIZER:
215  // Skip this here (done below)
216  break;
217  }
218  }
219  }
220  if (TanksWithFuel == 0) CurrentFuelTankPriority++; // No tanks at this priority, try next priority
221  }
222 
223  bool FuelStarved = Starved;
224  Starved = true;
225 
226  // Process Oxidizer tanks, if any
227  if (engine->GetType() == FGEngine::etRocket) {
228  while ((TanksWithOxidizer == 0) && (CurrentOxidizerTankPriority <= numTanks)) {
229  for (unsigned int i=0; i<engine->GetNumSourceTanks(); i++) {
230  unsigned int TankId = engine->GetSourceTank(i);
231  FGTank* Tank = Tanks[TankId];
232  unsigned int TankPriority = Tank->GetPriority();
233  if (TankPriority != 0) {
234  switch(Tank->GetType()) {
235  case FGTank::ttFUEL:
236  // Skip this here (done above)
237  break;
238  case FGTank::ttOXIDIZER:
239  hasOxTanks = true;
240  if (Tank->GetContents() > Tank->GetUnusable() && Tank->GetSelected() && TankPriority == CurrentOxidizerTankPriority) {
241  TanksWithOxidizer++;
242  if (TanksWithFuel > 0) Starved = false;
243  FeedListOxi.push_back(TankId);
244  }
245  break;
246  }
247  }
248  }
249  if (TanksWithOxidizer == 0) CurrentOxidizerTankPriority++; // No tanks at this priority, try next priority
250  }
251  }
252 
253  bool OxiStarved = Starved;
254 
255  engine->SetStarved(FuelStarved || (hasOxTanks && OxiStarved)); // Tanks can be refilled, so be sure to reset engine Starved flag here.
256 
257  // No fuel or fuel/oxidizer found at any priority!
258 // if (Starved) return;
259  if (FuelStarved || (hasOxTanks && OxiStarved)) return;
260 
261  double FuelToBurn = engine->CalcFuelNeed(); // How much fuel does this engine need?
262  double FuelNeededPerTank = FuelToBurn / TanksWithFuel; // Determine fuel needed per tank.
263  for (unsigned int i=0; i<FeedListFuel.size(); i++) {
264  Tanks[FeedListFuel[i]]->Drain(FuelNeededPerTank);
265  }
266 
267  if (engine->GetType() == FGEngine::etRocket) {
268  double OxidizerToBurn = engine->CalcOxidizerNeed(); // How much fuel does this engine need?
269  double OxidizerNeededPerTank = 0;
270  if (TanksWithOxidizer > 0) OxidizerNeededPerTank = OxidizerToBurn / TanksWithOxidizer; // Determine fuel needed per tank.
271  for (unsigned int i=0; i<FeedListOxi.size(); i++) {
272  Tanks[FeedListOxi[i]]->Drain(OxidizerNeededPerTank);
273  }
274  }
275 
276 }

280 bool FGPropulsion::GetSteadyState(void)
281 {
282  double currentThrust = 0, lastThrust = -1;
283  int steady_count = 0, j = 0;
284  bool steady = false;
285  bool TrimMode = FDMExec->GetTrimStatus();
286  double TimeStep = FDMExec->GetDeltaT();
287 
288  vForces.InitMatrix();
289  vMoments.InitMatrix();
290 
291  if (!FGModel::Run(false)) {
292  FDMExec->SetTrimStatus(true);
293  // This is a time marching algorithm so it needs a non-zero time step to
294  // reach a steady state.
295  in.TotalDeltaT = 0.5;
296 
297  for (unsigned int i=0; i<numEngines; i++) {
298  steady=false;
299  steady_count=0;
300  j=0;
301  while (!steady && j < 6000) {
302  Engines[i]->Calculate();
303  lastThrust = currentThrust;
304  currentThrust = Engines[i]->GetThrust();
305  if (fabs(lastThrust-currentThrust) < 0.0001) {
306  steady_count++;
307  if (steady_count > 120) {
308  steady=true;
309  }
310  } else {
311  steady_count=0;
312  }
313  j++;
314  }
315  vForces += Engines[i]->GetBodyForces(); // sum body frame forces
316  vMoments += Engines[i]->GetMoments(); // sum body frame moments
317  }
318 
319  FDMExec->SetTrimStatus(TrimMode);
320  in.TotalDeltaT = TimeStep;
321 
322  return false;
323  } else {
324  return true;
325  }
326 }

330 void FGPropulsion::InitRunning(int n)
331 {
332  if (n >= 0) { // A specific engine is supposed to be initialized
333 
334  if (n >= (int)GetNumEngines() ) {
335  throw(string("Tried to initialize a non-existent engine!"));
336  }
337 
338  in.ThrottleCmd[n] = in.ThrottlePos[n] = 1; // Set the throttle command and position
339  in.MixtureCmd[n] = in.MixturePos[n] = 1; // Set the mixture command and position
340 
341  GetEngine(n)->InitRunning();
342  GetSteadyState();
343 
344  } else if (n < 0) { // -1 refers to "All Engines"
345 
346  for (unsigned int i=0; i<GetNumEngines(); i++) {
347  in.ThrottleCmd[i] = in.ThrottlePos[i] = 1; // Set the throttle command and position
348  in.MixtureCmd[i] = in.MixturePos[i] = 1; // Set the mixture command and position
349  GetEngine(i)->InitRunning();
350  }
351 
352  GetSteadyState();
353  }

358 bool FGPropulsion::Load(Element* el)
359 {
360  FGModelLoader ModelLoader(this);
361 
362  Debug(2);
363  ReadingEngine = false;
364  double FuelDensity = 6.0;
365 
366  Name = "Propulsion Model: " + el->GetAttributeValue("name");
367 
368  // Perform base class Pre-Load
369  if (!FGModel::Upload(el, true))
370  return false;
371 
372  // Process tank definitions first to establish the number of fuel tanks
373 
374  Element* tank_element = el->FindElement("tank");
375  while (tank_element) {
376  Tanks.push_back(new FGTank(FDMExec, tank_element, numTanks));
377  if (Tanks.back()->GetType() == FGTank::ttFUEL) {
378  FuelDensity = Tanks[numFuelTanks]->GetDensity();
379  numFuelTanks++;
380  }
381  else if (Tanks.back()->GetType() == FGTank::ttOXIDIZER) numOxiTanks++;
382  else {cerr << "Unknown tank type specified." << endl; return false;}
383  numTanks++;
384  tank_element = el->FindNextElement("tank");
385  }
386  numSelectedFuelTanks = numFuelTanks;
387  numSelectedOxiTanks = numOxiTanks;
388 
389  ReadingEngine = true;
390  Element* engine_element = el->FindElement("engine");
391  while (engine_element) {
392  if (!ModelLoader.Open(engine_element)) return false;
393 
394  try {
395  // Locate the thruster definition
396  Element* thruster_element = engine_element->FindElement("thruster");
397  if (!thruster_element || !ModelLoader.Open(thruster_element))
398  throw("No thruster definition supplied with engine definition.");
399 
400  if (engine_element->FindElement("piston_engine")) {
401  HavePistonEngine = true;
402  if (!IsBound) bind();
403  Element *element = engine_element->FindElement("piston_engine");
404  Engines.push_back(new FGPiston(FDMExec, element, numEngines, in));
405  } else if (engine_element->FindElement("turbine_engine")) {
406  HaveTurbineEngine = true;
407  if (!IsBound) bind();
408  Element *element = engine_element->FindElement("turbine_engine");
409  Engines.push_back(new FGTurbine(FDMExec, element, numEngines, in));
410  } else if (engine_element->FindElement("turboprop_engine")) {
411  HaveTurboPropEngine = true;
412  if (!IsBound) bind();
413  Element *element = engine_element->FindElement("turboprop_engine");
414  Engines.push_back(new FGTurboProp(FDMExec, element, numEngines, in));
415  } else if (engine_element->FindElement("rocket_engine")) {
416  HaveRocketEngine = true;
417  if (!IsBound) bind();
418  Element *element = engine_element->FindElement("rocket_engine");
419  Engines.push_back(new FGRocket(FDMExec, element, numEngines, in));
420  } else if (engine_element->FindElement("electric_engine")) {
421  HaveElectricEngine = true;
422  if (!IsBound) bind();
423  Element *element = engine_element->FindElement("electric_engine");
424  Engines.push_back(new FGElectric(FDMExec, element, numEngines, in));
425  } else {
426  cerr << engine_element->ReadFrom() << " Unknown engine type" << endl;
427  return false;
428  }
429  } catch (std::string& str) {
430  cerr << endl << fgred << str << reset << endl;
431  return false;
432  }
433 
434  numEngines++;
435 
436  engine_element = el->FindNextElement("engine");
437  }
438 
439  CalculateTankInertias();
440 
441  if (el->FindElement("dump-rate"))
442  DumpRate = el->FindElementValueAsNumberConvertTo("dump-rate", "LBS/MIN");
443  if (el->FindElement("refuel-rate"))
444  RefuelRate = el->FindElementValueAsNumberConvertTo("refuel-rate", "LBS/MIN");
445 
446  unsigned int i;
447  for (i=0; i<Engines.size(); i++) {
448  Engines[i]->SetFuelDensity(FuelDensity);
449  }
450 
451 
452  PostLoad(el, FDMExec);
453 
454  return true;
455 }

591 const FGMatrix33& FGPropulsion::CalculateTankInertias(void)
592 {
593  size_t size = Tanks.size();
594 
595  if (size == 0) return tankJ;
596 
597  tankJ.InitMatrix();
598 
599  for (unsigned int i=0; i<size; i++) {
600  tankJ += FDMExec->GetMassBalance()->GetPointmassInertia( lbtoslug * Tanks[i]->GetContents(),
601  Tanks[i]->GetXYZ());
602  tankJ(1,1) += Tanks[i]->GetIxx();
603  tankJ(2,2) += Tanks[i]->GetIyy();
604  tankJ(3,3) += Tanks[i]->GetIzz();
605  }
606 
607  return tankJ;

663 void FGPropulsion::SetCutoff(int setting)
664 {
665  bool bsetting = setting == 0 ? false : true;
666 
667  if (ActiveEngine < 0) {
668  for (unsigned i=0; i<Engines.size(); i++) {
669  switch (Engines[i]->GetType()) {
670  case FGEngine::etTurbine:

760 void FGPropulsion::DoRefuel(double time_slice)
761 {
762  unsigned int i;
763 
764  double fillrate = RefuelRate / 60.0 * time_slice;
765  int TanksNotFull = 0;
766 
767  for (i=0; i<numTanks; i++) {
768  if (Tanks[i]->GetPctFull() < 99.99) ++TanksNotFull;
769  }
770 
771  // adds fuel equally to all tanks that are not full
772  if (TanksNotFull) {
773  for (i=0; i<numTanks; i++) {
774  if (Tanks[i]->GetPctFull() < 99.99)
775  Transfer(-1, i, fillrate/TanksNotFull);
776  }
777  }

783 {

788  if (Tanks[i]->GetContents() > Tanks[i]->GetStandpipe()) ++TanksDumping;
789  }
